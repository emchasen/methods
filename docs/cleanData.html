read_horizon <- function(){
  horizon <- list()
  for(i in 1:length(counties)){
    filename = paste0("data/", counties[i], "_SSURGO_chorizon.csv")
    horizon[[i]] <- read.csv(file = filename, na.strings = c(" ", "", "NA", "NaN"))
  }
  horizon <- do.call(rbind.data.frame, horizon)
}
